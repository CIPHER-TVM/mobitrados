Note in the ajax layout, we don't use app.min.js, instead using ajax.min.js
app.min.js is still used, which is loaded later on ajax content

Ajax Content

* Again as a reminder, you should use the ajax starter page so that you understand correctly

Here is a simple example of ajax content:

<p>Hi, this content is loaded via ajax</p>

<script>
  var script = [
    '../js/app.min.js'
  ]
  App.loadPlugins(null, script).then(() => {

    // your script here

  }).then(() => App.stopLoading())
</script>
app.min.js must be loaded on every page (except the login page) so that the template runs as it should

Loading JS & CSS Files

To add new styles to a page, you can simply insert "style" tags:

<style>
  .welcome {
    font-weight: bold;
    color: blue;
  }
</style>

<p class="welcome">Hi, this content is loaded via ajax</p>

<script>
  var script = [
    '../js/app.min.js'
  ]
  App.loadPlugins(null, script).then(() => {

    // your script here

  }).then(() => App.stopLoading())
</script>

To load new css and js files, you should do it in the following way:

<p>Hi, this content is loaded via ajax</p>

<script>
  var plugins = [
    'path/to/your/style.css',
    'path/to/your/another_style.css',
    'path/to/your/script.js',
    'path/to/your/another_script.js',
  ]
  var script = [
    '../js/app.min.js'
  ]
  App.loadPlugins(plugins, script).then(() => {

    // your script here

  }).then(() => App.stopLoading())
</script>
Please note that, scripts loaded from "plugins" variable will be loaded and executed only once.
But sometimes, some scripts are related to a page and should be run everytime the page is loaded.
In that case you can put the script to "script" variable 
<p>Hi, this content is loaded via ajax</p>

<script>
  var plugins = [
    'path/to/your/plugin.css',
    'path/to/your/another_plugin.css',
    'path/to/your/pluginscript.js',
    'path/to/your/another_pluginscript.js',
  ]
  var script = [
    '../js/app.min.js',
    'path/to/your/script.js'
  ]
  App.loadPlugins(plugins, script).then(() => {

    // your script here

  }).then(() => App.stopLoading())
</script>

Loading JS & CSS Files Synchronously

loadPlugins use javascript Promise which perform request asynchronously.
For some cases you need to load other plugins in sequence, you can add true to third parameter
this will make Promise work called "Async/await" which means perform request synchronously 
<p>Hi, this content is loaded via ajax</p>

<script>
  var plugins = [
    'path/to/your/style.css',
    'path/to/your/another_style.css',
    'path/to/your/script1.js',
    'path/to/your/script2_that_required_script1.js',
    '../dist/js/app.min.js',
  ]
  var script = [
    '../js/app.min.js'
  ]
  App.loadPlugins(plugins, script, true).then(() => { // add 'true' to third parameter to perform request synchronously 

    // your script here

  }).then(() => App.stopLoading())
</script>
